class Incubadora extends StatefulWidget {
  final String rpcUrl;
  final String contractAddress;
  final String abi;
  final EthPrivateKey credentials;
  final String userAddress;
  final String firmaId;
  final int nidoId;
  final int canastaId;
  final int assignedNumber;

  const Incubadora({
    super.key,
    required this.rpcUrl,
    required this.contractAddress,
    required this.abi,
    required this.credentials,
    required this.userAddress,
    required this.firmaId,
    required this.nidoId,
    required this.canastaId,
    required this.assignedNumber,
  });

  @override
  State<Incubadora> createState() => _Incubadora();
}

class _Incubadora extends State<Incubadora> {
  late Web3Client ethClient;
  late DeployedContract contract;

  int? numeroAsignado;
  Duration restante = const Duration(minutes: 1);
  int participantesCount = 0;
  bool _mostrarMensajeBienvenida = true;

  String ganadorDireccion = '';
  int ganadorNumero = 0;
  bool mostrandoGanador = false;
  Timer? _timer;
  bool _procesoIniciado = false;

  int restanteLimpiando = 30;
  Timer? _limpiezaTimer;

  bool _enviandoHuevos = false;
  bool _abriendoHuevo = false;

  final List<StreamSubscription> _subs = [];

  @override
  void initState() {
    super.initState();
    ethClient = Web3Client(
      widget.rpcUrl,
      Client());

    numeroAsignado = widget.assignedNumber;
    _mostrarMensajeBienvenida = true;

    _initContract().then((_) {
      _actualizarContadorParticipantes();
      _iniciarPollingGanador();
      Timer.periodic(const Duration(seconds: 5), (timer) {
        if (!mounted || mostrandoGanador) {
          timer.cancel();
          return;
        }
        _actualizarContadorParticipantes();

      });
    });
  }

  Future<void> _actualizarContadorParticipantes() async {
    try {

      final resultCount = await _callWithRetry(() => ethClient.call(
        contract: contract,
        function: contract.function('jugadoresEnCanasta'),
        params: [BigInt.from(widget.nidoId), BigInt.from(widget.canastaId)],
      ), operationName: 'Obtener contador participantes');

      final int count = (resultCount[0] as BigInt).toInt();

      if (mounted) {
        setState(() {
          participantesCount = count;
        });
      }
    } catch (e) {"Error al actualizar el contador de participantes.";}
  }
  Timer? _pollingTimer;
  bool _pollingActivo = true;

  void _iniciarPollingGanador() {
    const intervalo = Duration(seconds: 15);
    const timeoutMaximo = Duration(minutes: 10);

    final horaInicio = DateTime.now();

    _pollingTimer = Timer.periodic(intervalo, (timer) async {
      if (!_pollingActivo || DateTime.now().difference(horaInicio) > timeoutMaximo) {
        timer.cancel();
        return;
      }

      try {

        final estadoResult = await _callWithRetry<List<dynamic>>(() => ethClient.call(
          contract: contract,
          function: contract.function("estadoCanasta"),
          params: [BigInt.from(widget.nidoId), BigInt.from(widget.canastaId)],
        ), operationName: 'Polling estadoCanasta', maxRetries: 2);

        final estado = estadoResult[0] as String;

        if (estado == "llena" && !_procesoIniciado) {
          timer.cancel();
          if (mounted) {
            _iniciarFaseEnviandoHuevos();
          }
        }
      } catch (e) {"Error al realizar el polling";}
    });
  }

  @override
  void dispose() {
    for (final s in _subs) {
      s.cancel();
    }
    ethClient.dispose();
    _timer?.cancel();
    _limpiezaTimer?.cancel();
    _pollingActivo = false;
    _pollingTimer?.cancel();
    super.dispose();
  }

  Future<void> _initContract() async {
    contract = DeployedContract(
      ContractAbi.fromJson(widget.abi, 'Huevobit'),
      EthereumAddress.fromHex(widget.contractAddress),
    );

    _listenEvents();
  }

  void _iniciarFaseEnviandoHuevos() {
    if (_timer != null && _timer!.isActive) {
      _timer!.cancel();
    }

    setState(() {
      restante = const Duration(minutes: 1);
      _enviandoHuevos = true;
      _abriendoHuevo = false;
      mostrandoGanador = false;
      _procesoIniciado = true;
    });

    _timer = Timer.periodic(const Duration(seconds: 1), (timer) {
      if (!mounted) {
        timer.cancel();
        return;
      }
      setState(() {
        final newSeconds = restante.inSeconds - 1;
        restante = Duration(seconds: newSeconds > 0 ? newSeconds : 0);
      });

      if (restante.inSeconds <= 0) {
        timer.cancel();
        _iniciarFaseAbriendoHuevo();
      }
    });
  }

  void _iniciarFaseAbriendoHuevo() {
    if (_timer != null && _timer!.isActive) {
      _timer!.cancel();
    }

    setState(() {
      restante = const Duration(seconds: 30);
      _enviandoHuevos = false;
      _abriendoHuevo = true;
    });

    _timer = Timer.periodic(const Duration(seconds: 1), (timer) {
      if (!mounted) {
        timer.cancel();
        return;
      }
      setState(() {
        final newSeconds = restante.inSeconds - 1;
        restante = Duration(seconds: newSeconds > 0 ? newSeconds : 0);
      });

      if (restante.inSeconds <= 0) {
        timer.cancel();
        _mostrarGanadorYSalir();
      }
    });
  }

  void _mostrarGanadorYSalir() {
    if (!mounted) return;

    setState(() {
      mostrandoGanador = true;
      _abriendoHuevo = false;
      _iniciarLimpieza();
    });
  }

  void _listenEvents() {
    final evGanadorDeclarado = contract.event('GanadorDeclarado');
    final evNumeroAsignado = contract.event('NumeroAsignado');

    _subs.add(
        ethClient.events(
            FilterOptions.events(contract: contract, event: evNumeroAsignado)
        ).listen((event) async {
          if (!mounted) return;
          try {
            final decoded = evNumeroAsignado.decodeResults(event.topics!, event.data!);

            final EthereumAddress jugador = decoded[0] as EthereumAddress;
            final BigInt nidoIdEvent = decoded[1] as BigInt;
            final BigInt canastaIdEvent = decoded[2] as BigInt;
            final int numeroAsignadoEvent = (decoded[3] as BigInt).toInt();

            if (nidoIdEvent.toInt() == widget.nidoId &&
                canastaIdEvent.toInt() == widget.canastaId &&
                jugador.hex.toLowerCase() == widget.credentials.address.hex.toLowerCase()) {

              if (mounted) {
                setState(() {
                  numeroAsignado = numeroAsignadoEvent;
                });
              }
              await _actualizarContadorParticipantes();
            }
          } catch (error) {"Error al procesar el evento de Número Asignado.";}
        })
    );

    _subs.add(
        ethClient.events(
            FilterOptions.events(contract: contract, event: evGanadorDeclarado)
        ).listen((e) async {
          if (!mounted) return;
          try {
            final decoded = evGanadorDeclarado.decodeResults(e.topics!, e.data!);

            final BigInt canastaIdEvent = BigInt.parse(
              e.topics![1].toString().replaceFirst('0x', ''),
              radix: 16,
            );

            if (canastaIdEvent != BigInt.from(widget.canastaId)) {
              return;
            }

            final EthereumAddress ganador = decoded[1] as EthereumAddress;
            final int numeroGanador = (decoded[3] as BigInt).toInt();

            if (mounted) {
              setState(() {

                ganadorDireccion = ganador.hex;
                ganadorNumero = numeroGanador;
              });
            }

            await _actualizarContadorParticipantes();

            if (!_procesoIniciado) {
              _iniciarFaseEnviandoHuevos();
            }

          } catch (error) {"Error al procesar el evento de número ganador";}
        })
    );
  }

  Future<T> _callWithRetry<T>(Future<T> Function() rpcCall, {String operationName = 'Operación RPC', int maxRetries = 3}) async {
    int attempt = 0;
    while (true) {
      try {
        attempt++;
        return await rpcCall();
      } catch (e) {
        if (attempt >= maxRetries) {
          rethrow;
        }
        await Future.delayed(Duration(seconds: 2 * attempt));
      }
    }
  }

  String _short(String address) {
    if (address.length <= 10) return address;
    return '${address.substring(0, 6)}...${address.substring(address.length - 4)}';
  }

  String _formatDurationHMS(Duration d) {
    final m = d.inMinutes.remainder(60).toString().padLeft(2, '0');
    final s = d.inSeconds.remainder(60).toString().padLeft(2, '0');
    return '$m:$s';
  }

  void _iniciarLimpieza() {
    restanteLimpiando = 30;
    _limpiezaTimer = Timer.periodic(const Duration(seconds: 1), (timer) {
      if (!mounted) {
        timer.cancel();
        return;
      }
      setState(() {
        restanteLimpiando--;
      });

      if (restanteLimpiando <= 0) {
        timer.cancel();
        Navigator.of(context).pushAndRemoveUntil(
          MaterialPageRoute(builder: (_) => const NidoScreen()),
              (Route<dynamic> route) => false,
        );
      }
    });
  }

  Widget _buildAnimacionLottie() {
    String assetPath = 'assets/animations/incubadora.json';

    if (_abriendoHuevo) {
      assetPath = 'assets/animations/abriendo_huevo.json';
    }

    return SizedBox(
      height: 300,
      width: 300,
      child: Lottie.asset(
        assetPath,
        repeat: true,
        fit: BoxFit.contain,
      ),
    );
  }
