pragma solidity ^0.8.24;

import "@openzeppelin/contracts/access/Ownable.sol";
import "@openzeppelin/contracts/utils/ReentrancyGuard.sol";
import "@openzeppelin/contracts/utils/cryptography/ECDSA.sol";
import "@openzeppelin/contracts/utils/cryptography/MessageHashUtils.sol";

contract Huevobit is Ownable, ReentrancyGuard {
    using ECDSA for bytes32;
    using MessageHashUtils for bytes32;

    // ---------- Estructuras ----------
    struct Jugador {
        address direccion;
        uint8 numero;                  
    }

    struct Canasta {
        Jugador[] jugadores;           
        uint256 balance;
    }
    

    // ---------- Constantes ----------
    address public constant devWallet = 0x26db29498Df486236a307c8dFC96e2BD45EEC8A5;
    uint256 public constant MAX_JUGADORES = 3;  
    uint256 public constant devFeeWei = 1e14;

    // ---------- Estados ----------
    mapping(uint256 => mapping(uint256 => mapping(uint8 => bool))) private numeroUsado; 
    mapping(bytes32  => bool) public hashesUsados;                  
    mapping(bytes32  => uint256) public hashToNido;             
    mapping(uint256  => uint256) public precioNido;             
    mapping(uint256 => mapping(uint256 => Canasta)) public canastasPorNido; 
    mapping(address => uint256) public ultimaActualizacion;
    mapping(address => uint256) public nonces;

    // ---------- Eventos ----------
    event NumeroAsignado(address indexed jugador, uint256 indexed nidoId, uint256 indexed canastaId, uint8 numero);
    event GanadorDeclarado(uint256 indexed canastaId, address ganador, uint256 premio, uint8 numero);
    event HashGenerado(address indexed usuario, uint256 indexed idNido, bytes32 indexed hashPago);

    // ---------- Constructor ----------
    constructor() Ownable(msg.sender) {
        // Precios iniciales
        precioNido[1] = 1e16;  // 0.01 BNB
        precioNido[2] = 2e16;  // 0.02 BNB
        precioNido[3] = 3e16;  // 0.03 BNB
    }

    // ---------- Gestión de precios ----------
    function setPrecio(uint256 idNido, uint256 precioWei) external onlyOwner {
        require(precioWei > 0, "precio cero");
        precioNido[idNido] = precioWei;        
    }

    // ---------- Pago y registro de hash ----------
    function pagarEntrada(uint256 idNido, bytes32 hashPago, uint256 nonce) external payable {
        uint256 precio = precioNido[idNido];
        require(precio > 0, "Nido inexistente");
        require(msg.value == precio, "Monto incorrecto");
        require(hashPago != bytes32(0), "Hash no validado");
        require(nonces[msg.sender] == nonce, "Nonce no validado");
        require(hashToNido[hashPago] == 0, "Hash ya registrado");        
        hashToNido[hashPago] = idNido; 
        nonces[msg.sender]++;

        emit HashGenerado(msg.sender, idNido, hashPago);
    }    

    // ---------- Consulta de nonce ----------
    function obtenerNonce(address usuario) external view returns (uint256) {
        return nonces[usuario];
    }

    // ---------- Entrada a la canasta ----------
    function entrarCanasta(
        string calldata idUsuario,
        bytes calldata firmaId,
        bytes32 hashPago,
        uint256 idNido,
        uint256 canastaId       
    ) external nonReentrant {
        require(hashToNido[hashPago] == idNido, "Hash no corresponde al nido");
        require(!hashesUsados[hashPago], "Hash ya utilizado");
        hashesUsados[hashPago] = true; 

        Canasta storage canasta = canastasPorNido[idNido][canastaId];

        // Validar firma del ID de usuario
        bytes32 mensaje = keccak256(abi.encodePacked(idUsuario)).toEthSignedMessageHash();
        require(mensaje.recover(firmaId) == msg.sender, "Firma no validada");

        // Asignar número único siempre distinto
        uint8 numero = _generarNumero(msg.sender, idNido, canastaId, canasta);
        canasta.jugadores.push(Jugador(msg.sender, numero));
        canasta.balance += precioNido[idNido];
        emit NumeroAsignado(msg.sender, idNido, canastaId, numero);

        // Si se llena la canasta -> iniciar sorteo
        if (canasta.jugadores.length >= MAX_JUGADORES) {
            _iniciarSorteo(idNido, canastaId);        
        }             
    }   

    // ---------- Sorteo ----------
    function _iniciarSorteo(uint256 idNido, uint256 canastaId) internal {
        Canasta storage canasta = canastasPorNido[idNido][canastaId];
        require(canasta.jugadores.length == MAX_JUGADORES, "Canasta incompleta");
        
        uint256 premio = (canasta.balance * 90) / 100;
        uint256 devFee = devFeeWei;

        uint256 indiceGanador = _indiceAleatorio(canasta);
        address ganador = canasta.jugadores[indiceGanador].direccion;
        uint8 numeroGanador = canasta.jugadores[indiceGanador].numero;

        // Transferencias
        payable(ganador).transfer(premio);
        if (devFee > 0) payable(devWallet).transfer(devFee);

        emit GanadorDeclarado(canastaId, ganador, premio, numeroGanador);       

        // Reiniciar canasta
        for (uint8 i = 1; i <= 10; i++) {
            numeroUsado[idNido][canastaId][i] = false;
        }
        delete canasta.jugadores;
        canasta.balance = 0;
    }      

    // ---------- Generar número único ----------
    function _generarNumero(
        address usuario,
        uint256 idNido,
        uint256 canastaId,
        Canasta storage canasta
    ) internal returns (uint8) {
        require(canasta.jugadores.length < MAX_JUGADORES, "Canasta llena");

        uint256 libres = 10 - canasta.jugadores.length;
        uint256 idx = 1 + (uint256(
        keccak256(abi.encodePacked(usuario, canasta.jugadores.length, canastaId, address(this)))
        ) % libres);


        uint8 numero = 1;
        uint8 contador = 1;
        for (uint8 i = 1; i <= 10; i++) {
            if (!numeroUsado[idNido][canastaId][i]) {
                if (contador == idx) { 
                    numero = i; 
                    break; 
                }
                contador++;
            }
        }

        numeroUsado[idNido][canastaId][numero] = true;
        return numero;
    }

    // ---------- Elige al ganador aleatorio ----------
    function _indiceAleatorio(Canasta storage canasta) internal view returns (uint256) {
        return uint256(
            keccak256(abi.encodePacked(block.timestamp, canasta.jugadores.length, block.prevrandao))
        ) % canasta.jugadores.length; // ⚠️ En producción: cambiar por Chainlink VRF
    }   

    // ---------- Retiro de residuales ----------
    function retirarFondos() external onlyOwner {
        payable(devWallet).transfer(address(this).balance);
    }    

    // ---------- Contador ----------
    function jugadoresEnCanasta(uint256 idNido, uint256 canastaId) external view returns (uint256) {
        return canastasPorNido[idNido][canastaId].jugadores.length;
    }     

    function obtenerJugadores(uint256 idNido, uint256 canastaId) 
    external view returns (address[] memory, uint8[] memory) 
    {
    Canasta storage canasta = canastasPorNido[idNido][canastaId];
    address[] memory direcciones = new address[](canasta.jugadores.length);
    uint8[] memory numeros = new uint8[](canasta.jugadores.length);
    
    for (uint i = 0; i < canasta.jugadores.length; i++) {
        direcciones[i] = canasta.jugadores[i].direccion;
        numeros[i] = canasta.jugadores[i].numero;
    }
    
    return (direcciones, numeros);
    }

    // ---------- Control de tiempo entre actualizaciones ----------
    function puedeActualizar(address usuario) external view returns (bool, uint256) {
        uint256 ultima = ultimaActualizacion[usuario];
        uint256 tiempoEspera = 1 minutes;
        uint256 tiempoRestante = ultima + tiempoEspera > block.timestamp ? 
        (ultima + tiempoEspera) - block.timestamp : 0;
    
        return (block.timestamp >= ultima + tiempoEspera, tiempoRestante);
    }

    // ---------- Guardar actualización ----------
    function _registrarActualizacion(address usuario) external {
        ultimaActualizacion[usuario] = block.timestamp;
    }

    // ---------- Verificación del estado de una canasta ----------
function estadoCanasta(uint256 idNido, uint256 canastaId) external view returns (string memory) {
    Canasta storage canasta = canastasPorNido[idNido][canastaId];

    if (canasta.jugadores.length >= MAX_JUGADORES) {
        return "llena";
    } else if (canasta.jugadores.length > 0) {
        return "en curso";
    } else {
        return "disponible";
    }   
    }

}
