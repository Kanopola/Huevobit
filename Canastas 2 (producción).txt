class CanastasScreen extends StatefulWidget {
  final String rpcUrl;
  final String contractAddress;
  final String abi;
  final String userAddress;
  final int idNido;
  final String idUsuario;
  final String firmaId;
  final String hashPago;

  const CanastasScreen({
    super.key,
    required this.rpcUrl,
    required this.contractAddress,
    required this.abi,
    required this.userAddress,
    required this.idNido,
    required this.idUsuario,
    required this.firmaId,
    required this.hashPago,
  });

  @override
  State<CanastasScreen> createState() => _CanastasScreenState();
}

class _CanastasScreenState extends State<CanastasScreen> {
  late Web3Client ethClient;
  late DeployedContract contract;
  late EthPrivateKey _credentials;

  Timer? _timerCooldown;
  bool _puedeActualizar = true;
  int _segundosRestantes = 0;

  bool _mostrarMensajeBienvenida = true;

  final FlutterSecureStorage _secureStorage = const FlutterSecureStorage();

  final Map<int, int> _participantesPorCanasta = {};
  final Map<int, String> _estadosPorCanasta = {};
  bool _refrescando = false;

  @override
  void initState() {
    super.initState();
    ethClient = Web3Client(widget.rpcUrl, http.Client());
    _initContract();
    _loadCredentials();
    _cargarCooldownPersistente();
    SystemChrome.setEnabledSystemUIMode(SystemUiMode.immersiveSticky);
    _mostrarMensajeBienvenida = true;

  }

  Future<void> _initContract() async {
    contract = DeployedContract(
      ContractAbi.fromJson(widget.abi, "Huevobit"),
      EthereumAddress.fromHex(widget.contractAddress),
    );
  }

  Future<void> _loadCredentials() async {
    try {
      final pk = await _secureStorage.read(key: 'privateKey');
      if (pk == null || pk.isEmpty) {
        throw Exception('No hay privateKey en Secure Storage');
      }
      _credentials = EthPrivateKey.fromHex(pk);
    } catch (e) {
      throw Exception(
          'No se pudieron cargar las credenciales. Por favor, inténtalo nuevamente.'
      );
    }
  }

  /// Carga desde storage el cooldown guardado
  Future<void> _cargarCooldownPersistente() async {
    final restanteStr = await _secureStorage.read(key: 'cooldownRestante');
    if (restanteStr != null) {
      final restante = int.tryParse(restanteStr) ?? 0;
      if (restante > 0) {
        setState(() {
          _puedeActualizar = false;
          _segundosRestantes = restante;
        });
        _iniciarTimer();
      }
    }
  }

  /// Refresca todas las canastas y muestra loading mientras se actualiza
  Future<void> _refrescarCanastas() async {
    if (!_puedeActualizar) return;

    // 1️⃣ Verificar si puede actualizar
    try {
      final function = contract.function("puedeActualizar");
      final result = await ethClient.call(
        contract: contract,
        function: function,
        params: [_credentials.address],
      );

      final puede = result[0] as bool;
      final tiempoRestante = (result[1] as BigInt).toInt();

      if (!puede) {
        setState(() {
          _puedeActualizar = false;
          _segundosRestantes = tiempoRestante;
        });
        _secureStorage.write(
            key: 'cooldownRestante', value: tiempoRestante.toString());
        _iniciarTimer();
        return;
      }
    } catch (e) {
      throw Exception(
          'No se pudieron actualizar las canastas en este momento. Por favor, inténtalo nuevamente más tarde.'
      );
    }

    if (!mounted) return;

    // 2️⃣ Confirmación del usuario
    final confirmar = await showDialog<bool>(
      context: context,
      builder: (context) => AlertDialog(
        title: const Text('Refrescar Canastas'),
        content: const Text(
            '¿Estás seguro? Evita abusos para que el nodo no te bloquee la IP.'),
        actions: [
          TextButton(
            onPressed: () => Navigator.of(context).pop(false),
            child: const Text('Cancelar'),
          ),
          TextButton(
            onPressed: () => Navigator.of(context).pop(true),
            child: const Text('Refrescar'),
          ),
        ],
      ),
    );

    if (confirmar != true) return;

    // 3️⃣ Mostrar círculo de carga
    if (mounted) setState(() => _refrescando = true);

    // 4️⃣ Actualizar todas las canastas
    try {
      for (int canastaId = 1; canastaId <= 100; canastaId++) {
        try {
          // Obtener estado
          final estadoResult = await ethClient.call(
            contract: contract,
            function: contract.function("estadoCanasta"),
            params: [BigInt.from(widget.idNido), BigInt.from(canastaId)],
          );
          final estado = estadoResult.isNotEmpty ? estadoResult[0] as String : "desconocido";

          // Obtener número de participantes
          final participantesResult = await ethClient.call(
            contract: contract,
            function: contract.function("jugadoresEnCanasta"),
            params: [BigInt.from(widget.idNido), BigInt.from(canastaId)],
          );
          final participantesCount = participantesResult.isNotEmpty
              ? (participantesResult[0] as BigInt).toInt()
              : 0;

          if (mounted) {
            setState(() {
              _participantesPorCanasta[canastaId] = participantesCount;
              _estadosPorCanasta[canastaId] = estado;
            });
          }
        } catch (e) {
          throw Exception(
              'No se pudieron actualizar algunas canastas. Por favor, inténtalo nuevamente más tarde.'
          );
        }


        // Pequeño delay para refresco visual del loading
        await Future.delayed(const Duration(milliseconds: 30));
      }
    } finally {
      // 5️⃣ Ocultar loading
      if (mounted) setState(() => _refrescando = false);
    }

    // 6️⃣ Iniciar cooldown
    setState(() {
      _puedeActualizar = false;
      _segundosRestantes = 60;
    });
    _secureStorage.write(
        key: 'cooldownRestante', value: _segundosRestantes.toString());
    _iniciarTimer();
  }


  void _iniciarTimer() {
    _timerCooldown?.cancel();
    _timerCooldown = Timer.periodic(const Duration(seconds: 1), (timer) {
      if (_segundosRestantes <= 1) {
        timer.cancel();
        setState(() {
          _puedeActualizar = true;
          _segundosRestantes = 0;
        });
        _secureStorage.delete(key: 'cooldownRestante');
      } else {
        setState(() => _segundosRestantes--);
        _secureStorage.write(
            key: 'cooldownRestante', value: _segundosRestantes.toString());
      }
    });
  }


  Future<void> _entrarCanasta(int canastaId) async {
    try {
        // 1. Verificar si tu dirección ya está en la canasta
      final result = await ethClient.call(
        contract: contract,
        function: contract.function("obtenerJugadores"),
        params: [
          BigInt.from(widget.idNido),
          BigInt.from(canastaId),
        ],
      );

      final direccionesRaw = result[0] as List;
      final direcciones = direccionesRaw.map((addr) {
        if (addr is EthereumAddress) return addr;
        if (addr is String) return EthereumAddress.fromHex(addr);
        throw Exception("Tipo de dirección inesperado: $addr");
      }).toList();

      bool estaDentro = direcciones.any((addr) => addr.hex.toLowerCase() == _credentials.address.hex.toLowerCase());

      if (estaDentro) {
        await _mostrarDialogo(
          "Información",
          "⏳ Ya estás dentro de esta canasta.",
          Colors.blue,
        );
        return;
      }


      // 2. Obtener cantidad de jugadores para mostrar en diálogo
      final participantesResult = await ethClient.call(
        contract: contract,
        function: contract.function("jugadoresEnCanasta"),
        params: [BigInt.from(widget.idNido), BigInt.from(canastaId)],
      );

      final participantesCount = participantesResult.isNotEmpty
          ? (participantesResult[0] as BigInt).toInt()
          : 0;

      if (mounted) {
        setState(() {
          _participantesPorCanasta[canastaId] = participantesCount;
        });
      }

      // 3. Mostrar diálogo de confirmación
      if (!mounted) return;
      final entrarConfirmado = await showDialog<bool>(
        context: context,
        builder: (context) => AlertDialog(
          title: Text('Canasta $canastaId'),
          content: Text('Participantes: $participantesCount/3\n¿Deseas entrar a esta canasta?'),
          actions: [
            TextButton(
              onPressed: () => Navigator.of(context).pop(false),
              child: const Text('Cancelar', style: TextStyle(color: Colors.red)),
            ),
            TextButton(
              onPressed: () => Navigator.of(context).pop(true),
              child: const Text('Entrar', style: TextStyle(color: Colors.green)),
            ),
          ],
        ),
      );

      if (entrarConfirmado != true) return;

      // 4. Enviar transacción para entrar
      final function = contract.function("entrarCanasta");
      final firma = Uint8List.fromList(hexToBytes(widget.firmaId));

      final txHash = await ethClient.sendTransaction(
        _credentials,
        Transaction.callContract(
          contract: contract,
          function: function,
          parameters: [
            widget.idUsuario,
            firma,
            hexToBytes(widget.hashPago),
            BigInt.from(widget.idNido),
            BigInt.from(canastaId),
          ],
          gasPrice: EtherAmount.fromBigInt(EtherUnit.gwei, BigInt.from(20)),
          maxGas: 500000,
        ),
        chainId: 31337,
      );

      // 5. Esperar confirmación de la transacción
      TransactionReceipt? receipt;
      final timeoutAt = DateTime.now().add(const Duration(seconds: 120));

      while (DateTime.now().isBefore(timeoutAt)) {
        try {
          receipt = await ethClient.getTransactionReceipt(txHash);

          // Actualizar contador de participantes
          final tmp = await ethClient.call(
            contract: contract,
            function: contract.function("jugadoresEnCanasta"),
            params: [BigInt.from(widget.idNido), BigInt.from(canastaId)],
          );

          final tmpCount = tmp.isNotEmpty ? (tmp[0] as BigInt).toInt() : 0;
          if (mounted) {
            setState(() {
              _participantesPorCanasta[canastaId] = tmpCount;
            });
          }

          if (receipt != null && receipt.status == true) break;
          if (receipt != null && receipt.status == false) throw Exception("Transacción fallida");
        } catch (e) {
          throw Exception(
            "⏳ La red está tardando más de lo normal.\n\n"
                "Tu transacción puede seguir pendiente en la blockchain.\n"
                "Verifica el historial en tu billetera más tarde o con el hash: $txHash",
          );
        }
        await Future.delayed(const Duration(seconds: 2));
      }

      if (receipt == null) throw Exception("Timeout esperando confirmación de transacción");
      if (receipt.status == false) throw Exception("Transacción fallida");

      // ✅ 6. ESPERAR EVENTO NumeroAsignado ANTES DE NAVEGAR
      int? assignedNumber;
      bool eventoRecibido = false;

      // Listener para el evento NumeroAsignado
      late StreamSubscription subscription;
      subscription = ethClient
          .events(FilterOptions.events(
        contract: contract,
        event: contract.event("NumeroAsignado"),
      )).listen((event) {
        try {
          final decoded = contract.event("NumeroAsignado").decodeResults(event.topics!, event.data!);

          final EthereumAddress jugador = decoded[0] as EthereumAddress;
          final BigInt nidoIdEvent = decoded[1] as BigInt;
          final BigInt canastaIdEvent = decoded[2] as BigInt;
          final int numeroAsignadoEvent = (decoded[3] as BigInt).toInt();

          // Verificar si es nuestro evento
          if (nidoIdEvent.toInt() == widget.idNido && // ← NUEVA VERIFICACIÓN
              canastaIdEvent.toInt() == canastaId &&
              jugador.hex.toLowerCase() == _credentials.address.hex.toLowerCase()) {
            eventoRecibido = true;
            assignedNumber = numeroAsignadoEvent;
            subscription.cancel();
          }
        } catch (e) {"Error decodificando evento Número Asignado";}
      });

      // ✅ 7. Fallback: consultar logs
      Future<void> checkLogsForNumber() async {
        try {
          final logs = await ethClient.getLogs(
            FilterOptions.events(
              contract: contract,
              event: contract.event("NumeroAsignado"),
              fromBlock: BlockNum.exact(receipt!.blockNumber as int),
              toBlock: BlockNum.exact(receipt.blockNumber as int),
            ),
          );

          for (final log in logs) {
            final decoded = contract.event("NumeroAsignado")
                .decodeResults(log.topics!, log.data!);

            final EthereumAddress jugador = EthereumAddress.fromHex(
              '0x${log.topics![1].toString().substring(26)}',
            );
            final int canastaIdEvent = BigInt.parse(
              log.topics![3].toString().replaceFirst('0x', ''),
              radix: 16,
            ).toInt();
            final int numeroAsignadoEvent = (decoded[3] as BigInt).toInt();

            if (canastaIdEvent == canastaId &&
                jugador.hex.toLowerCase() == _credentials.address.hex.toLowerCase()) {
              eventoRecibido = true;
              assignedNumber = numeroAsignadoEvent;
              subscription.cancel();
              break;
            }
          }
        } catch (e) {"Error decodificando evento Número Asignado";}
      }

      // ✅ 8. Esperar máximo 10 segundos por el evento
      final eventTimeout = DateTime.now().add(const Duration(seconds: 10));

      while (DateTime.now().isBefore(eventTimeout) && !eventoRecibido) {
        // Intentar fallback después de 3 segundos si no llegó el evento
        if (DateTime.now().isAfter(DateTime.now().add(const Duration(seconds: 3))) ){
          await checkLogsForNumber();
        }

        if (!eventoRecibido) {
          await Future.delayed(const Duration(milliseconds: 500));
        }
      }

      // ✅ 9. Si no se recibió el evento, intentar consultar directamente
      if (!eventoRecibido) {
        try {
          final result = await ethClient.call(
            contract: contract,
            function: contract.function("numeroDeJugador"),
            params: [
              _credentials.address,
              BigInt.from(widget.idNido),
              BigInt.from(canastaId),
            ],
          );

          if (result.isNotEmpty) {
            assignedNumber = (result[0] as BigInt).toInt();
            eventoRecibido = true;
          }
        } catch (e) {"Error al consultar el número del jugador";}
      }

      if (!eventoRecibido || assignedNumber == null) {
        throw Exception("No se pudo obtener el número asignado. Verifica la transacción más tarde.");
      }        // ✅ 10. Navegar a Incubadora con el número asignado (o 0 como fallback)
        if (!mounted) return;

        Navigator.push(
          context,
          MaterialPageRoute(
            builder: (_) =>
                Incubadora(
                  rpcUrl: widget.rpcUrl,
                  contractAddress: contract.address.hex,
                  abi: widget.abi,
                  credentials: _credentials,
                  userAddress: widget.userAddress,
                  firmaId: widget.firmaId,
                  nidoId: widget.idNido,
                  canastaId: canastaId,
                  assignedNumber: assignedNumber!,
                ),
          ),
        );

    } catch (e) {
      // Manejar errores específicos
      String mensaje = "Error al entrar a la canasta";

      if (e.toString().contains("Hash ya utilizado")) {
        mensaje = "⚠️ Hash ya utilizado. Genera uno nuevo en un Nido.";
      } else if (e.toString().contains("Hash no corresponde al Nido")) {
        mensaje = "❌ Hash no válido para este Nido";
      } else if (e.toString().contains("Firma no validada")) {
        mensaje = "❌ Firma inválida";
      } else if (e.toString().contains("Jugador no encontrado")) {
        mensaje = "⏳ La canasta está en proceso de sorteo. Busca otra canasta o espera unos minutos.";
      } else if (e.toString().contains("No se pudo obtener el número")) {
        mensaje = "⏳ Tiempo de espera agotado. Verifica más tarde si tu número fue asignado.";
      } else if (e.toString().contains("Canasta llena")) {
        mensaje = "❌ La canasta se llenó justo ahora. Elige otra canasta.";
      } else {
        mensaje = "Error: ${e.toString()}";
      }
      await _mostrarDialogo("Información", mensaje, Colors.blue);
    }

  }

  /// Muestra un AlertDialog simple con color de fondo y título coloreado.
  /// Verifica mounted antes de abrir el diálogo para eliminar los warnings.
  Future<void> _mostrarDialogo(String titulo, String mensaje,
      Color colorFondo) async {
    if (!mounted) return;
    await showDialog(
      context: context,
      builder: (context) =>
          AlertDialog(
            backgroundColor: colorFondo,
            // fondo sólido
            title: Text(titulo, style: const TextStyle(color: Colors.white)),
            // título en blanco
            content: Text(mensaje, style: const TextStyle(color: Colors.white)),
            // contenido en blanco
            actions: [
              TextButton(
                onPressed: () {
                  if (mounted) Navigator.of(context).pop();
                },
                child: const Text('Aceptar'),
              ),
            ],
          ),
    );
  }

  String shortHash(String hash) {
    if (hash.length <= 10) return hash;
    return '${hash.substring(0, 6)}...${hash.substring(hash.length - 4)}';
  }


  Color _getBorderColor(int canastaId) {
    final estado = _estadosPorCanasta[canastaId] ?? "disponible";
    switch (estado) {
      case "llena":
        return Colors.red;
      case "en curso":
        return Colors.orange;
      case "disponible":
        return Colors.green;
      default:
        return Colors.green;
    }
  }
